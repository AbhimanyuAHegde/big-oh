190 	/**
191 	 * confirm that event listeners are notified as expected
192 	 */
193 	@Test
194 	public void testEventNotification()
195 	{
196 
197 		// define a private listener class that can be used to count number of
198 		// events received
199 		class InvocationCountingListener implements MaximalCliqueFinderEventListener<JungVertex>
200 		{
201 
202 			private int cliqueFoundCount = 0;
203 			private int stepCompletedCount = 0;
204 
205 			public void cliqueFound(Set<JungVertex> clique)
206 			{
207 				cliqueFoundCount++;
208 			}
209 
210 			public void stepCompleted(long numStepsCompleted, long expectedTotalSteps)
211 			{
212 				stepCompletedCount++;
213 			}
214 
215 			public int getCliqueFoundCount()
216 			{
217 				return cliqueFoundCount;
218 			}
219 
220 			public int getStepCompletedCount()
221 			{
222 				return stepCompletedCount;
223 			}
224 
225 		}
226 
227 		// create listeners
228 		InvocationCountingListener listener1 = new InvocationCountingListener();
229 		InvocationCountingListener listener2 = new InvocationCountingListener();
230 
231 		// attach listeners
232 		cliqueFinder.addListener(listener1);
233 		cliqueFinder.addListener(listener2);
234 
235 		// find cliques with listeners attached to cliqueFinder
236 		Set<Set<JungVertex>> maximalCliques = cliqueFinder.findMaximalCliques(graph, 2);
237 		assertFalse(maximalCliques.isEmpty());
238 
239 		// confirm that stepCompleted(..) was called a reasonable number of
240 		// times
241 		assertTrue(listener1.getStepCompletedCount() >= graph.getVertexCount());
242 		assertTrue(listener2.getStepCompletedCount() >= graph.getVertexCount());
243 
244 		// confirm that cliqueFound(..) was called the expected number of times
245 		assertEquals(maximalCliques.size(), listener1.getCliqueFoundCount());
246 		assertEquals(maximalCliques.size(), listener2.getCliqueFoundCount());
247 	}

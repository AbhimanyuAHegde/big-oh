31  /**
32   * Provides a base level of abstraction for graph algorithms. The underlying
33   * implementation of this class might use the JUNG framework or, alternatively,
34   * might provide a custom view of SQL data that adheres to the interface.
35   * 
36   * @author dwingate
37   * @version Dec 29, 2009
38   * @param <V>
39   */
40  public interface Graph<V extends Vertex>
41  {
42  	/**
43  	 * Add a vertex to the graph.
44  	 * 
45  	 * @param v
46  	 * @throws IllegalArgumentException
47  	 *             Thrown if v is null.
48  	 */
49  	public void addVertex(V v) throws IllegalArgumentException;
50  
51  	/**
52  	 * Connects two vertices in the graph with an edge. The implementing class
53  	 * may choose to implement either a directed or undirected edge. Depending
54  	 * on that implementation choice, the post conditions of this method will
55  	 * vary.
56  	 * 
57  	 * @param fromVertex
58  	 * @param toVertex
59  	 * @throws IllegalArgumentException
60  	 *             if either {@link Vertex} argument is null.
61  	 */
62  	public void addEdge(V fromVertex, V toVertex) throws IllegalArgumentException;
63  
64  	/**
65  	 * @return Returns the set of all vertices contained in the graph.
66  	 */
67  	public Set<V> getAllVertices();
68  
69  	/**
70  	 * @return Returns an iterator that traverses the set of all vertices
71  	 *         contained in the graph.
72  	 */
73  	public Iterator<V> getAllVerticesIterator();
74  
75  	/**
76  	 * @return Returns a count of all vertices in the graph.
77  	 */
78  	public long getVertexCount();
79  
80  	/**
81  	 * @param v
82  	 * @return Returns the set of all neighbors to the vertex.
83  	 */
84  	public Set<V> getAllNeighbors(V v);
85  }

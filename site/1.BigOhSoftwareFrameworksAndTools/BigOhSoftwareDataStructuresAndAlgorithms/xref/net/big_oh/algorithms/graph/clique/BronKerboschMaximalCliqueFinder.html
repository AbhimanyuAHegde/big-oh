43  /**
44   * An implementation of the Bron–Kerbosch algorithm for finding all maximal
45   * cliques in a graph.
46   * 
47   * <pre>
48   *    BronKerbosch1(R,P,X):
49   *      if P and X are both empty:
50   *          report R as a maximal clique
51   *      for each vertex v in P:
52   *          BronKerbosch1(R ⋃ {v}, P ⋂ N(v), X ⋂ N(v))
53   *          P := P \ {v}
54   *          X := X ⋃ {v}
55   * </pre>
56   * 
57   * @author dwingate
58   * @version Dec 30, 2009
59   * 
60   * @param <V>
61   */
62  public class BronKerboschMaximalCliqueFinder<V extends Vertex> implements MaximalCliqueFinder<V>
63  {
64  
65  	private static final Log logger = LogFactory.getLog(BronKerboschMaximalCliqueFinder.class);
66  
67  	private final Collection<MaximalCliqueFinderEventListener<V>> eventListeners;
68  
69  	public BronKerboschMaximalCliqueFinder()
70  	{
71  		// Use CopyOnWriteArrayList to support dynamic listeners w/o incurring
72  		// synchronization overhead
73  		eventListeners = new CopyOnWriteArrayList<MaximalCliqueFinderEventListener<V>>();
74  	}
75  
76  	public void addListener(MaximalCliqueFinderEventListener<V> listener)
77  	{
78  		// Note, no need to synchronize since eventListeners is an instance of
79  		// CopyOnWriteArrayList
80  		eventListeners.add(listener);
81  	}

83  	public Set<Set<V>> findMaximalCliques(Graph<V> g, int minCliqueSize) throws IllegalArgumentException
84  	{
85  
86  		Duration d = new Duration(this.getClass());
87  
88  		if (g == null)
89  		{
90  			throw new IllegalArgumentException("The Graph argument must not be null.");
91  		}
92  		if (minCliqueSize < 2)
93  		{
94  			throw new IllegalArgumentException("The minCliqueSize argument must not be less than two.");
95  		}
96  
97  		// for the first step of the BronKerbosch algorithm, use a Vertex
98  		// iterator instead of fetching the entire collection. For sparse
99  		// graphs, use of the iterator avoids creating unnecessarily large
100 		// P collections.
101 		Set<Set<V>> maximalCliques = new HashSet<Set<V>>();
102 
103 		long numVertices = g.getVertexCount();
104 		int completedVertexCounter = 0;
105 		for (Iterator<V> vertexIter = g.getAllVerticesIterator(); vertexIter.hasNext();)
106 		{
107 			V v = vertexIter.next();
108 
109 			Set<V> r = new HashSet<V>();
110 			r.add(v);
111 
112 			Set<V> p = g.getAllNeighbors(v);
113 
114 			Set<V> x = g.getAllNeighbors(v);
115 
116 			// process maximal cliques discovered for Vertex v
117 			processMaximalCliqueDiscoveries(maximalCliques, bronKerbosch1(r, p, x, g, minCliqueSize, 1));
118 
119 			// count the completion of a step in the algorithm
120 			completedVertexCounter++;
121 
122 			// signal step completion to listeners
123 			for (MaximalCliqueFinderEventListener<V> listener : eventListeners)
124 			{
125 				try
126 				{
127 					listener.stepCompleted(completedVertexCounter, numVertices);
128 				}
129 				catch (RuntimeException re)
130 				{
131 					logger.error(re.getMessage(), re);
132 				}
133 			}
134 
135 			// log current progress of the operation?
136 			if (completedVertexCounter % (numVertices / 100) == 0)
137 			{
138 				logger.info("Executing " + this.getClass().getSimpleName() + ".findMaximalCliques(..) ... " + (int) ((double) completedVertexCounter / (double) numVertices * 100.0) + "% complete.");
139 			}
140 		}
141 
142 		d.stop("findMaximalCliques");
143 
144 		logger.info("Discovered " + maximalCliques.size() + " maximal cliques with size of at least " + minCliqueSize + ".");
145 
146 		return maximalCliques;
147 
148 	}

150 	private Set<Set<V>> bronKerbosch1(Set<V> r, Set<V> p, Set<V> x, Graph<V> g, int minCliqueSize, int searchDepth)
151 	{
152 
153 		if (logger.isDebugEnabled())
154 		{
155 			logger.debug("Depth:" + searchDepth + " - R:" + r.size() + " - P:" + p.size() + " - X:" + x.size());
156 		}
157 
158 		// base case ...
159 		// if P and X are both empty, then we're done
160 		if (p.isEmpty() && x.isEmpty())
161 		{
162 			if (r.size() >= minCliqueSize)
163 			{
164 				// the maximal clique contained in r is big enough to report
165 				return Collections.singleton(r);
166 			}
167 			else
168 			{
169 				// the maximal clique contained in r is too small
170 				return Collections.emptySet();
171 			}
172 		}
173 
174 		// recursive case ...
175 		Set<Set<V>> maximalCliques = new HashSet<Set<V>>();
176 
177 		Set<V> pCopy = new HashSet<V>(p);
178 		for (V v : pCopy)
179 		{
180 
181 			// rPrime = R ⋃ {v}
182 			Set<V> rPrime = new HashSet<V>(r);
183 			rPrime.add(v);
184 
185 			// pPrime = P ⋂ N(v)
186 			Set<V> pPrime = new HashSet<V>(p);
187 			pPrime.retainAll(g.getAllNeighbors(v));
188 
189 			// xPrime = X ⋂ N(v)
190 			Set<V> xPrime = new HashSet<V>(x);
191 			xPrime.retainAll(g.getAllNeighbors(v));
192 
193 			maximalCliques.addAll(bronKerbosch1(rPrime, pPrime, xPrime, g, minCliqueSize, searchDepth + 1));
194 
195 			// P := P \ {v}
196 			p.remove(v);
197 
198 			// X := X ⋃ {v}
199 			x.add(v);
200 
201 		}
202 
203 		return maximalCliques;
204 
205 	}

207 	/**
208 	 * This helper method merges two sets of maximal cliques, claiming newly
209 	 * discovered maximal cliques and adding them to the set of well known
210 	 * maximal cliques.
211 	 * 
212 	 * @param wellKnownMaximalCliques
213 	 * @param maximalCliqueDiscoveries
214 	 */
215 	private void processMaximalCliqueDiscoveries(Set<Set<V>> wellKnownMaximalCliques, Set<Set<V>> maximalCliqueDiscoveries)
216 	{
217 		for (Set<V> maximalCliqueDiscovery : maximalCliqueDiscoveries)
218 		{
219 
220 			if (wellKnownMaximalCliques.add(maximalCliqueDiscovery))
221 			{
222 				// this is the first discovery of the maximal clique ...
223 
224 				// notify listeners
225 				for (MaximalCliqueFinderEventListener<V> listener : eventListeners)
226 				{
227 					try
228 					{
229 						listener.cliqueFound(maximalCliqueDiscovery);
230 					}
231 					catch (RuntimeException re)
232 					{
233 						logger.error(re.getMessage(), re);
234 					}
235 				}
236 			}
237 
238 		}
239 	}

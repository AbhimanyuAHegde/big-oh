62  	public AStarSearchResult<SearchNodeType> doSearch(SearchNodeType startNode)
63  	{
64  
65  		Duration searchDuration = new Duration();
66  
67  		// create priority queue
68  		SortedSet<SearchNodeType> prioritySet = new TreeSet<SearchNodeType>(new Comparator<SearchNodeType>()
69  		{
70  
71  			public int compare(SearchNodeType o1, SearchNodeType o2)
72  			{
73  				switch (searchType)
74  				{
75  				case MIN:
76  					return Double.compare(getF(o1), getF(o2));
77  				case MAX:
78  					return -1 * Double.compare(getF(o1), getF(o2));
79  				default:
80  					throw new RuntimeException("Unexpected search type: " + searchType);
81  				}
82  			}
83  
84  		});
85  
86  		// enqueue the start node
87  		prioritySet.add(startNode);
88  
89  		// declare tracking member variables
90  		int numSearchNodesGenerated = 0;
91  		int numSearchNodesConsidered = 0;
92  		int maxPossibleBranchingFactor = 1;
93  
94  		// search for a goal state
95  		SearchNodeType goalNode = null;
96  		while (!prioritySet.isEmpty())
97  		{
98  
99  			// Remove the best candidate node from the queue
100 			SearchNodeType candidateSearchNode = prioritySet.first();
101 			prioritySet.remove(candidateSearchNode);
102 			numSearchNodesConsidered++;
103 
104 			// get the next search node candidates
105 			Collection<SearchNodeType> nextSearchNodes = nextNodesGenerator.getNextSearchNodes(candidateSearchNode);
106 
107 			// do some record keeping
108 			numSearchNodesGenerated += nextSearchNodes.size();
109 			maxPossibleBranchingFactor = Math.max(maxPossibleBranchingFactor, nextSearchNodes.size());
110 
111 			if (candidateSearchNode.isGoalState())
112 			{
113 				// sanity check
114 				assert (nextSearchNodes.isEmpty());
115 
116 				// found an optimal solution
117 				goalNode = candidateSearchNode;
118 				break;
119 			}
120 			else
121 			{
122 				// enqueue all next search nodes
123 				Duration enqueueDuration = new Duration();
124 
125 				prioritySet.addAll(nextSearchNodes);
126 
127 				if (logger.isDebugEnabled())
128 				{
129 					logger.debug("Enqueued " + nextSearchNodes.size() + " A* search nodes in " + enqueueDuration.stop() + " milliseconds.");
130 				}
131 			}
132 
133 		}
134 
135 		// return the search results
136 		AStarSearchResult<SearchNodeType> results = new AStarSearchResult<SearchNodeType>(goalNode, numSearchNodesGenerated, calculateEffectiveBranchingFactor(goalNode.getNodeDepth(), numSearchNodesConsidered, maxPossibleBranchingFactor));
137 
138 		logger.debug("Completed an A* search in " + searchDuration.stop() + " milliseconds.");
139 		logger.debug("Number of nodes generated: " + results.getNumSearchNodesGenerated());
140 		logger.debug("Depth of goal node: " + results.getGoalNode().getNodeDepth());
141 		logger.debug("Effective branching factor: " + results.getEfectiveBranchingFactor());
142 
143 		return results;
144 
145 	}

162 	static double calculateEffectiveBranchingFactor(long solutionDepth, long numSearchNodesConsidered, int maxPossibleBranchingFactor)
163 	{
164 		// sanity check
165 		if (solutionDepth >= numSearchNodesConsidered)
166 		{
167 			throw new IllegalArgumentException("The solutionDepth argument must always be less than the numSearchNodesConsidered argument.");
168 		}
169 
170 		// edge case
171 		if (solutionDepth == 0)
172 		{
173 			return 1.0;
174 		}
175 
176 		double ebf = Math.pow((double) numSearchNodesConsidered, (1.0 / solutionDepth));
177 		return Math.min(ebf, maxPossibleBranchingFactor);
178 	}

46  /**
47   * The invocation handler for a {@link Proxy} that stands in for a Driver-created SQL {@link Connection} so that
48   * all Statements created on the Connection can be observed.
49   */
50  class JdbcObserverProxyConnectionInvocationHandler implements InvocationHandler
51  {
52  
53  	private static final Log logger = LogFactory.getLog(JdbcObserverProxyConnectionInvocationHandler.class);
54  
55  	private final Connection originalConnection;
56  
57  	private final List<JDBCEventListener> listeners;
58  
59  	JdbcObserverProxyConnectionInvocationHandler(Connection originalConnection, List<JDBCEventListener> listeners)
60  	{
61  		this.originalConnection = originalConnection;
62  
63  		// Create a new list of listeners to side-step the possibility of
64  		// ConcurrentModificationException
65  		this.listeners = Collections.unmodifiableList(new ArrayList<JDBCEventListener>(listeners));
66  	}
67  
68  	public Object invoke(Object proxy, Method method, Object[] args) throws Throwable
69  	{
70  
71  		// intercept calls that create statements
72  		if (Statement.class.isAssignableFrom(method.getReturnType()))
73  		{
74  
75  			StatementInstantiationEvent event = new StatementInstantiationEvent((Connection) proxy);
76  
77  			for (JDBCEventListener listener : listeners)
78  			{
79  				try
80  				{
81  					listener.statementRequested(event);
82  				}
83  				catch (RuntimeException rte)
84  				{
85  					logger.error(rte);
86  				}
87  			}
88  
89  			Statement newStatement = (Statement) method.invoke(originalConnection, args);
90  
91  			// Create proxy for the returned statement object
92  			String hardCodedSqlForStatement = (newStatement instanceof PreparedStatement && args != null && args.length > 0 && args[0] instanceof String) ? args[0].toString() : null;
93  			boolean newStatementInterfacesContainsMethodReturnTypeClass = Arrays.asList(newStatement.getClass().getInterfaces()).contains(method.getReturnType());
94  			Class<?>[] interfaces = new Class<?>[(newStatement.getClass().getInterfaces().length + ((newStatementInterfacesContainsMethodReturnTypeClass) ? 0 : 1))];
95  			System.arraycopy(newStatement.getClass().getInterfaces(), 0, interfaces, 0, newStatement.getClass().getInterfaces().length);
96  			if (!newStatementInterfacesContainsMethodReturnTypeClass)
97  			{
98  				interfaces[newStatement.getClass().getInterfaces().length] = method.getReturnType();
99  			}
100 			Statement statementProxy = (Statement) Proxy.newProxyInstance(this.getClass().getClassLoader(), interfaces, new JdbcObserverProxyStatementInvocationHandler(newStatement, listeners, hardCodedSqlForStatement));
101 
102 			for (JDBCEventListener listener : listeners)
103 			{
104 				try
105 				{
106 					listener.statementInstantiated(event, statementProxy);
107 				}
108 				catch (RuntimeException rte)
109 				{
110 					logger.error(rte);
111 				}
112 			}
113 
114 			return statementProxy;
115 		}
116 
117 		// otherwise, just invoke the method on the underlying
118 		// originalConnection
119 		return method.invoke(originalConnection, args);
120 
121 	}

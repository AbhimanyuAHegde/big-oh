12   * A JDBCEventListener that prevents the execution of dangerous SQL by throwing
13   * an instance of SQLException.

18  public class JDBCSafetyListener extends JDBCEventAdapter

20  
21  	@Override
22  	public void beforeSQLExecution(SQLExecutionEvent event) throws SQLException
23  	{
24  		
25  		for (String sqlCommand : event.getSqlCommands())
26  		{
27  			
28  			// prevent execution of DELETE statements with no WHERE clause
29  			if (RegExpHelper.matches(sqlCommand.toUpperCase(), "^DELETE\\s.*") && !RegExpHelper.contains(sqlCommand.toUpperCase(), "WHERE"))
30  			{
31  				throw new SQLException("Cowardly refusing to excute a DELETE statement that has no WHERE clause.  Use the TRUNCATE command if you really want to remove all data from a table?");
32  			}
33  			
34  			// prevent execution of statements that look suspiciously like an SQL injection attack via inserted ";"
35  			if (RegExpHelper.contains(sqlCommand, ";\\s*\\w+")) {
36  				throw new SQLException("Cowardly refusing to excute a SQL statement that smells suspiciously like an SQL injection attack.");
37  			}
38  			
39  		}
40  	
41  	}

51  /**
52   * <p>
53   * A proxy class that intercepts JDBC {@link Driver} calls in order to expose
54   * those calls as observable events.
55   * </p>
56   * <p>
57   * To avoid the proliferation of child classes, this class provides only a
58   * no-arg constructor that reads a list of listener classes from
59   * jdbcdriverproxy.properties.
60   * </p>
61   */
62  public final class JdbcDriverProxy implements Driver
63  {
64  
65  	private static final Log logger = LogFactory.getLog(JdbcDriverProxy.class);
66  
67  	// register the driver with DriverManager
68  	static
69  	{
70  		try
71  		{
72  			DriverManager.registerDriver(new JdbcDriverProxy());
73  		}
74  		catch (SQLException e)
75  		{
76  			logger.error(e);
77  		}
78  	}

85  	 * Constructs a JdbcDriverProxy using the listeners defined in
86  	 * jdbcdriverproxy.properties.
87  	 */
88  	public JdbcDriverProxy()

90  		this(buildListenersListFromPropertiesFile());
91  	}

93  	private static final List<? extends JDBCEventListener> buildListenersListFromPropertiesFile()
94  	{
95  		Properties props = PropertyUtil.loadProperties(JdbcDriverProxy.class.getSimpleName().toLowerCase());
96  
97  		List<JDBCEventListener> listeners = new ArrayList<JDBCEventListener>();
98  
99  		String[] listenerClassNames = props.getProperty(JDBCEventListener.class.getSimpleName() + "s", "").split(",");
100 
101 		for (String listenerClassName : listenerClassNames)
102 		{
103 
104 			if (!StringUtils.isBlank(listenerClassName))
105 			{
106 
107 				try
108 				{
109 					Class<?> listenerClass = Class.forName(listenerClassName);
110 					JDBCEventListener listener = (JDBCEventListener) listenerClass.newInstance();
111 					listeners.add(listener);
112 					logger.info("Registered a new JDBC event listener: " + listenerClass.toString());
113 				}
114 				catch (ClassNotFoundException cnfe)
115 				{
116 					logger.error("Failed to find a JDBC event listener class named '" + listenerClassName + "'", cnfe);
117 				}
118 				catch (InstantiationException e)
119 				{
120 					logger.error("Failed to instantiate a JDBC event listener class named '" + listenerClassName + "'", e);
121 				}
122 				catch (IllegalAccessException e)
123 				{
124 					logger.error(e);
125 				}
126 
127 			}
128 
129 		}
130 
131 		return listeners;
132 
133 	}

135 	/**
136 	 * Constructs a JdbcDriverProxy from a list of JDBCEventListener object.
137 	 * 
138 	 * @param listeners
139 	 */
140 	private JdbcDriverProxy(List<? extends JDBCEventListener> listeners)
141 	{
142 
143 		// Create a new list of listeners to side-step the possibility of
144 		// ConcurrentModificationException
145 		this.listeners = Collections.unmodifiableList(new ArrayList<JDBCEventListener>(listeners));
146 
147 		for (String driverClassName : getDriverClassNamesToAutoLoad())
148 		{
149 			logger.debug("Attempting to auto-load delegate Driver " + driverClassName + ".");
150 			try
151 			{
152 				Class.forName(driverClassName);
153 			}
154 			catch (ClassNotFoundException cnfe)
155 			{
156 				logger.warn("Failed to auto-load delegate Driver " + driverClassName + " because the class could not be found.");
157 			}
158 		}
159 
160 	}

163 	 * 
164 	 * @return The prefix that this ProxyDriver will expect to be pre-pended to
165 	 *         URL connection strings.
166 	 */
167 	private final String getJdbcObserverUrlPrefix()
168 	{
169 		return this.getClass().getSimpleName() + ":";
170 	}

172 	/**
173 	 * Translates the URL expected by this ProxyDriver to a URL expected by the
174 	 * delegateDriver.
175 	 * 
176 	 * @param url
177 	 *            A URL string that begins with the <code>String</code> returned
178 	 *            by <code>getJdbcObserverUrlPrefix()</code>
179 	 * @return
180 	 */
181 	private final String stripJdbcObserverPrefixFromUrl(String url)
182 	{
183 		if (!url.startsWith(getJdbcObserverUrlPrefix()))
184 		{
185 			return url;
186 		}
187 		else
188 		{
189 			return url.substring(getJdbcObserverUrlPrefix().length());
190 		}
191 	}

193 	/**
194 	 * 
195 	 * @return A <code>Set</code> of delegate driver names that you'd like to
196 	 *         use without having to explicitly call
197 	 *         <code>Class.forName(com.somepackage.SomeDriver);</code>.
198 	 */
199 	private Set<String> getDriverClassNamesToAutoLoad()
200 	{
201 		return JdbcDriverProxyHelper.getStandardDriversToAutoLoad();
202 	}

204 	private final Driver lookupDelegateDriver(String url) throws SQLException
205 	{
206 		if (!url.startsWith(getJdbcObserverUrlPrefix()))
207 		{
208 			logger.debug("Requested url '" + url + "' is not recognized by " + this.getClass().getName());
209 			return null;
210 		}
211 
212 		url = stripJdbcObserverPrefixFromUrl(url);
213 
214 		for (Driver delegateDriverCandidate : CollectionsUtil.toIterable(DriverManager.getDrivers()))
215 		{
216 			if (delegateDriverCandidate != this && delegateDriverCandidate.acceptsURL(url))
217 			{
218 				logger.debug("Selected driver " + delegateDriverCandidate.getClass().getName() + " to service url '" + url + "'");
219 				registerMostRecentDelegateDriver(mostRecentDelegateDriver);
220 				return delegateDriverCandidate;
221 			}
222 		}
223 
224 		logger.warn("Failed to find an appropriate delegate driver for url '" + url + "'");
225 		logger.warn("Maybe you need to load the delegate Driver class by calling 'Class.forName(com.somepackage.SomeDriver);' or by updating the overridden getDriverClassNamesToAutoLoad() method ?");
226 		return null;
227 
228 	}

230 	private final void registerMostRecentDelegateDriver(Driver driver)
231 	{
232 		if (this.mostRecentDelegateDriver != null && !this.mostRecentDelegateDriver.getClass().isAssignableFrom(driver.getClass()))
233 		{
234 			logger.warn("Encountered multiple delegate driver clas types (" + mostRecentDelegateDriver.getClass().getName() + " & " + driver.getClass().getName()
235 					+ ").  Using this proxy with multiple delegate types can cause surprising results to be returned by the getMajorVersion(), getMinorVersion() & jdbcCompliant() methods.");
236 		}
237 		this.mostRecentDelegateDriver = driver;
238 	}

242 		Driver delegateDriver = lookupDelegateDriver(url);
243 		return (delegateDriver == null) ? false : delegateDriver.acceptsURL(url);

246 	public Connection connect(String url, Properties connectionProperties) throws SQLException
247 	{
248 		Driver delegateDriver = lookupDelegateDriver(url);
249 
250 		if (delegateDriver == null)
251 		{
252 			// return null per the API guidelines
253 			return null;
254 		}
255 
256 		url = stripJdbcObserverPrefixFromUrl(url);
257 
258 		ConnectionInstantiationEvent connectionInstantiationEvent = new ConnectionInstantiationEvent(this, delegateDriver, url, connectionProperties);
259 
260 		for (JDBCEventListener listener : listeners)
261 		{
262 			try
263 			{
264 				listener.connectionRequested(connectionInstantiationEvent);
265 			}
266 			catch (RuntimeException rte)
267 			{
268 				logger.error(rte);
269 			}
270 		}
271 
272 		Connection newConnection = delegateDriver.connect(url, connectionProperties);
273 
274 		if (newConnection == null)
275 		{
276 			throw new SQLException("Failed to connect to url '" + url + "' using delegate driver " + delegateDriver.getClass().getName());
277 		}
278 
279 		// Create a proxy for the returned connection
280 		boolean newConnectionInterfacesContainsConnection = Arrays.asList(newConnection.getClass().getInterfaces()).contains(Connection.class);
281 		Class<?>[] interfaces = new Class<?>[(newConnection.getClass().getInterfaces().length + ((newConnectionInterfacesContainsConnection) ? 0 : 1))];
282 		System.arraycopy(newConnection.getClass().getInterfaces(), 0, interfaces, 0, newConnection.getClass().getInterfaces().length);
283 		if (!newConnectionInterfacesContainsConnection)
284 		{
285 			interfaces[newConnection.getClass().getInterfaces().length] = Connection.class;
286 		}
287 		Connection connectionProxy = (Connection) Proxy.newProxyInstance(this.getClass().getClassLoader(), interfaces, new JdbcObserverProxyConnectionInvocationHandler(newConnection, listeners));
288 
289 		for (JDBCEventListener listener : listeners)
290 		{
291 			try
292 			{
293 				listener.connectionInstantiated(connectionInstantiationEvent, connectionProxy);
294 			}
295 			catch (RuntimeException rte)
296 			{
297 				logger.error(rte);
298 			}
299 		}
300 
301 		return connectionProxy;
302 	}

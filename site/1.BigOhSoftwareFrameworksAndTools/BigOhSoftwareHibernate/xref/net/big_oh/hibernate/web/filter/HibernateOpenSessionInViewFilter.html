44  /**
45   * A reusable implementation of the classic "open session in view" pattern.
46   * 
47   * @author davewingate
48   * @version Sep 30, 2009
49   */
50  public class HibernateOpenSessionInViewFilter implements Filter
51  {
52  
53  	private static Log log = LogFactory.getLog(HibernateOpenSessionInViewFilter.class);
54  
55  	private SessionFactory sf;
56  
57  	public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws IOException, ServletException
58  	{
59  
60  		try
61  		{
62  			log.info("Starting a database transaction");
63  			sf.getCurrentSession().beginTransaction();
64  
65  			// Call the next filter (continue request processing)
66  			chain.doFilter(request, response);
67  
68  			// Commit and cleanup
69  			log.info("Committing the database transaction");
70  			sf.getCurrentSession().getTransaction().commit();
71  
72  		}
73  		catch (Throwable ex)
74  		{
75  			HibernateUtil.rollback(ex, sf, log);
76  
77  			// Let others handle it... maybe another interceptor for exceptions?
78  			throw new ServletException(ex);
79  		}
80  	}
81  
82  	public void init(FilterConfig filterConfig) throws ServletException
83  	{
84  		log.debug("Initializing filter...");
85  		log.debug("Obtaining SessionFactory from static HibernateUtil singleton");
86  		sf = HibernateUtil.getSessionFactory();
87  	}
88  
89  	public void destroy()
90  	{
91  	}

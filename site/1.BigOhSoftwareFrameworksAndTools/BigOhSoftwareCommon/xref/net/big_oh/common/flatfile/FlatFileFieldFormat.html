28  /**
29   * The FlatFileFieldFormat class represents the definition of a single field in
30   * a flat file record.
31   */
32  public class FlatFileFieldFormat
33  {
34  
35  	private final int fieldStartPosition;
36  	private final int fieldEndPosition;
37  	private final FlatFileFieldPaddingType paddingType;
38  	private final char intraFieldPaddingChar;
39  
40  	/**
41  	 * 
42  	 * @param fieldStartPosition
43  	 *            The 1-based start position of this field in the flat file
44  	 *            record.
45  	 * @param fieldEndPosition
46  	 *            The 1-based end position of this field in the flat file
47  	 *            record.
48  	 * @param paddingType
49  	 *            Defines how to handle values that are shorter than the field
50  	 *            length.
51  	 * @param intraFieldPaddingChar
52  	 *            Ignored if paddingType is set to
53  	 *            FlatFileFieldPaddingType.VARIABLE_WIDTH.
54  	 * @throws IllegalArgumentException
55  	 *             Thrown if fieldStartPosition is less than one or if
56  	 *             fieldEndPosition is less than fieldStartPosition.
57  	 */
58  	public FlatFileFieldFormat(int fieldStartPosition, int fieldEndPosition, FlatFileFieldPaddingType paddingType, char intraFieldPaddingChar) throws IllegalArgumentException
59  	{
60  		super();
61  
62  		if (fieldStartPosition < 1)
63  		{
64  			throw new IllegalArgumentException("The fieldStartPosition parameter cannot be less than one.");
65  		}
66  
67  		if (fieldEndPosition < fieldStartPosition)
68  		{
69  			throw new IllegalArgumentException("The fieldEndPosition parameter must not be less than the fieldStartPosition parameter.");
70  		}
71  
72  		this.fieldStartPosition = fieldStartPosition;
73  		this.fieldEndPosition = fieldEndPosition;
74  		this.paddingType = paddingType;
75  		this.intraFieldPaddingChar = intraFieldPaddingChar;
76  	}
77  
78  	int getFieldStartPosition()
79  	{
80  		return fieldStartPosition;
81  	}
82  
83  	int getFieldEndPosition()
84  	{
85  		return fieldEndPosition;
86  	}
87  
88  	FlatFileFieldPaddingType getPaddingType()
89  	{
90  		return paddingType;
91  	}
92  
93  	char getIntraFieldPaddingChar()
94  	{
95  		return intraFieldPaddingChar;
96  	}
97  
98  	int getFieldLength()
99  	{
100 		return getFieldEndPosition() - getFieldStartPosition() + 1;
101 	}
102 
103 	@Override
104 	public String toString()
105 	{
106 		StringBuffer sb = new StringBuffer();
107 		sb.append("Start Position: " + fieldStartPosition);
108 		sb.append(" -- End Position: " + fieldEndPosition);
109 		sb.append(" -- Padding Type: " + paddingType.name());
110 		sb.append(" -- Padding Char: " + intraFieldPaddingChar);
111 		return sb.toString();
112 	}

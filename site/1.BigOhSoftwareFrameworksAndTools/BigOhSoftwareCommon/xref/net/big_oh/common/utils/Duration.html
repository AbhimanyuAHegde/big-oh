32  /**
33   * The Duration class encapsulates the concept of measuring the time required to
34   * complete an activity, like the execution of one or more methods.
35   */
36  public class Duration
37  {
38  
39  	private static final Log defaultLogger = LogFactory.getLog(Duration.class);
40  
41  	private final long started;
42  	private final Log logger;
43  	private DURATION_STATUS status;
44  
45  	/**
46  	 * Constructs a default Duration instance using the Duration class to derive
47  	 * an appropriate logger.
48  	 */
49  	public Duration()
50  	{
51  		this(null);
52  	}
53  
54  	/**
55  	 * Constructs a default Duration instance using callingClass to derive an
56  	 * appropriate logger.
57  	 * 
58  	 * @param callingClass
59  	 */
60  	public Duration(Class<?> callingClass)
61  	{
62  		started = System.currentTimeMillis();
63  		logger = (callingClass == null) ? defaultLogger : LogFactory.getLog(callingClass);
64  		status = DURATION_STATUS.STARTED;
65  	}
66  
67  	/**
68  	 * Indicate that the duration has ended.
69  	 * 
70  	 * @return Length of duration in milliseconds.
71  	 */
72  	public long stop()
73  	{
74  		return stop(null);
75  	}
76  
77  	/**
78  	 * Indicate that the duration has ended, logging an appropriate
79  	 * informational message.
80  	 * 
81  	 * @param msg
82  	 *            A message to be logged at info level.
83  	 * @return Length of duration in milliseconds.
84  	 */
85  	public long stop(String msg)
86  	{
87  
88  		if (!status.equals(DURATION_STATUS.STARTED))
89  		{
90  			throw new RuntimeException("Duration is already been stopped.");
91  		}
92  
93  		status = DURATION_STATUS.STOPPED;
94  
95  		long durationInMilliseconds = System.currentTimeMillis() - started;
96  
97  		if (!StringUtils.isBlank(msg))
98  		{
99  			logger.info("Time for " + msg + ": " + durationInMilliseconds + " millisecoonds.");
100 		}
101 
102 		return durationInMilliseconds;
103 
104 	}
105 
106 	private enum DURATION_STATUS
107 	{
108 		STARTED, STOPPED;
109 	}

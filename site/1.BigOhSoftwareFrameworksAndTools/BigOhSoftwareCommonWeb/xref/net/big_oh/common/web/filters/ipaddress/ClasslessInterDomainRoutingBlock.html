28  import java.net.InetAddress;
29  import java.util.HashMap;
30  import java.util.Map;
31  
32  public class ClasslessInterDomainRoutingBlock
33  {
34  
35  	private final InetAddress baseIPAddress;
36  	private final int prefixLength;
37  
38  	public ClasslessInterDomainRoutingBlock(InetAddress baseIPAddress, int prefixLength)
39  	{
40  
41  		this.baseIPAddress = baseIPAddress;
42  		this.prefixLength = prefixLength;
43  
44  		if (prefixLength < 0 || prefixLength > (baseIPAddress.getAddress().length * 8))
45  		{
46  			throw new RuntimeException("The prefix length must be a value between 0 and " + (baseIPAddress.getAddress().length * 8) + " inclusive.");
47  		}
48  
49  	}
50  
51  	/**
52  	 * Tests to see whether a candidate address is a member of the routing
53  	 * block.
54  	 * 
55  	 * @param candidateAddress
56  	 * @return a boolean value indicating whether or not the candidateAddress is
57  	 *         a member of the routing block.
58  	 */
59  	public boolean isAddressMemberOfRoutingBlock(InetAddress candidateAddress)
60  	{
61  
62  		// Can't possibly be in the same block if addresses are of different
63  		// types (e.g. IPv4 and IPv6).
64  		if (!candidateAddress.getClass().isAssignableFrom(baseIPAddress.getClass()))
65  		{
66  			return false;
67  		}
68  
69  		Map<Integer, Integer> byteIndexToNumBitsMap = new HashMap<Integer, Integer>();
70  		int bitsToConsider = prefixLength;
71  		int byteCounter = 0;
72  		while (bitsToConsider > 0)
73  		{
74  			byteIndexToNumBitsMap.put(byteCounter++, Math.min(bitsToConsider, 8));
75  			bitsToConsider -= Math.min(bitsToConsider, 8);
76  		}
77  
78  		// Strategy here is to compare using XOR
79  		for (Integer byteIndex : byteIndexToNumBitsMap.keySet())
80  		{
81  
82  			Integer numBits = byteIndexToNumBitsMap.get(byteIndex);
83  
84  			int numPlacesToShift = Math.abs(8 - numBits);
85  
86  			byte baseByte = baseIPAddress.getAddress()[byteIndex];
87  			byte candidateByte = candidateAddress.getAddress()[byteIndex];
88  
89  			// use a mask to remove zero or more least significant bits that
90  			// should not be considered by subsequent the XOR operation
91  			int mask = -1 << numPlacesToShift;
92  			// (numPlacesToShift > 0) ? (int) (Math.pow(2, 8) - Math.pow(2,
93  			// numPlacesToShift)) : -1;
94  			int maskedBaseByte = baseByte & mask;
95  			int maskedCandidateByte = candidateByte & mask;
96  
97  			if ((maskedBaseByte ^ maskedCandidateByte) != 0)
98  			{
99  				return false;
100 			}
101 
102 		}
103 
104 		return true;
105 
106 	}

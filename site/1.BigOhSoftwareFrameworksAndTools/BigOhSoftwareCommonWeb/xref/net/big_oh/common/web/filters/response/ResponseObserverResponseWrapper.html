28  import java.io.IOException;
29  import java.io.PrintWriter;
30  
31  import javax.servlet.ServletOutputStream;
32  import javax.servlet.http.HttpServletResponse;
33  import javax.servlet.http.HttpServletResponseWrapper;
34  
35  public class ResponseObserverResponseWrapper extends HttpServletResponseWrapper
36  {
37  
38  	private final ResponseObserver observer;
39  	private final String filterName;
40  
41  	private ResponseObserverPrintWriter writer;
42  	private ResponseObserverOutputStream stream;
43  
44  	ResponseObserverResponseWrapper(HttpServletResponse response, ResponseObserver observer, String filterName)
45  	{
46  		super(response);
47  		this.observer = observer;
48  		this.filterName = filterName;
49  	}
50  
51  	@Override
52  	public ServletOutputStream getOutputStream() throws IOException
53  	{
54  		if (stream == null)
55  		{
56  			stream = new ResponseObserverOutputStream(super.getOutputStream(), observer);
57  		}
58  
59  		return stream;
60  	}
61  
62  	@Override
63  	public PrintWriter getWriter() throws IOException
64  	{
65  		if (writer == null)
66  		{
67  			writer = new ResponseObserverPrintWriter(super.getWriter(), observer);
68  		}
69  
70  		return writer;
71  	}
72  
73  	public String getFilterName()
74  	{
75  		return filterName;
76  	}
77  
78  }

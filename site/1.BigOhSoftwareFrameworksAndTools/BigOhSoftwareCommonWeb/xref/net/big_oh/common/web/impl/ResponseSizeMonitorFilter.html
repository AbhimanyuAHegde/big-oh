21  	private final ConcurrentHashMap<Thread, Long> threadToByteCountMap = new ConcurrentHashMap<Thread, Long>();
22  
23  	public void doBeforeResponseStarted(ServletRequest req, ServletResponse resp)
24  	{
25  		threadToByteCountMap.put(Thread.currentThread(), new Long(0));
26  	}
27  
28  	public void doAfterResponseFinished(ServletRequest req, ServletResponse resp)
29  	{
30  		long numBytesInResponse = threadToByteCountMap.get(Thread.currentThread());
31  		logger.info("RESPONSE -- FINISH -- Response for the requested url (" + getUrl(req) + ") contained about " + new DecimalFormat("###,##0.00").format(numBytesInResponse / 1024.0) + " kilobytes.");
32  
33  		// remove entry in threadToByteCountMap so that the Long value can be
34  		// garbage collected
35  		threadToByteCountMap.remove(Thread.currentThread());
36  	}
37  
38  	private String getUrl(ServletRequest req)
39  	{
40  		if (req instanceof HttpServletRequest)
41  		{
42  			return ((HttpServletRequest) req).getRequestURL().toString();
43  		}
44  		else
45  		{
46  			return "unknown; non HTTP";
47  		}
48  	}
49  
50  	public void doAfterCharsWritten(char[] charsWritten)
51  	{
52  		synchronized (threadToByteCountMap)
53  		{
54  			Long currentByteCount = threadToByteCountMap.get(Thread.currentThread());
55  			// Note: assuming that print writer is using the ISO-8859-1
56  			// encoding, which prints 1 byte per character.
57  			// Would need to update if you want to support counting response
58  			// bytes for other CharSets.
59  			threadToByteCountMap.put(Thread.currentThread(), new Long(currentByteCount + (charsWritten.length * 1)));
60  		}
61  	}
62  
63  	public void doAfterBytesWriten(byte[] bytesWritten)
64  	{
65  		synchronized (threadToByteCountMap)
66  		{
67  			Long currentByteCount = threadToByteCountMap.get(Thread.currentThread());
68  			threadToByteCountMap.put(Thread.currentThread(), new Long(currentByteCount + bytesWritten.length));

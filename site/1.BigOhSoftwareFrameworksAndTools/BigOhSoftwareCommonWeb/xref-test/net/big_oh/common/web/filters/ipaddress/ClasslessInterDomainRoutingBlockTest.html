36  import org.junit.After;

40  public class ClasslessInterDomainRoutingBlockTest
41  {
42  
43  	ClasslessInterDomainRoutingBlock leftEdge;
44  	ClasslessInterDomainRoutingBlock middle;
45  	ClasslessInterDomainRoutingBlock unusualPrefixLength;
46  	ClasslessInterDomainRoutingBlock rightEdge;
47  
48  	@Before
49  	public void setUp() throws Exception
50  	{
51  		leftEdge = new ClasslessInterDomainRoutingBlock(InetAddress.getByName("10.0.0.0"), 0);
52  		middle = new ClasslessInterDomainRoutingBlock(InetAddress.getByName("208.130.28.0"), 24);
53  		unusualPrefixLength = new ClasslessInterDomainRoutingBlock(InetAddress.getByName("208.130.28.255"), 25);
54  		rightEdge = new ClasslessInterDomainRoutingBlock(InetAddress.getByName("207.71.25.66"), 32);
55  	}
56  
57  	@After
58  	public void tearDown() throws Exception
59  	{
60  	}
61  
62  	@Test
63  	public void testIsAddressMemberOfRoutingBlock() throws UnknownHostException
64  	{
65  
66  		// all valid addresses are members of the "left edge" block
67  		assertTrue(leftEdge.isAddressMemberOfRoutingBlock(InetAddress.getByName("10.0.0.0")));
68  		assertTrue(leftEdge.isAddressMemberOfRoutingBlock(InetAddress.getByName("192.168.0.1")));
69  		assertTrue(leftEdge.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.0")));
70  		assertTrue(leftEdge.isAddressMemberOfRoutingBlock(InetAddress.getByName("127.0.0.1")));
71  		assertTrue(leftEdge.isAddressMemberOfRoutingBlock(InetAddress.getByName("207.71.25.66")));
72  
73  		// only certain addresses are members of the "middle" block
74  		// first test some of the myriad addresses that are within the block
75  		assertTrue(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.0")));
76  		assertTrue(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.1")));
77  		assertTrue(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.2")));
78  		assertTrue(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.3")));
79  		assertTrue(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.4")));
80  		assertTrue(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.252")));
81  		assertTrue(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.253")));
82  		assertTrue(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.254")));
83  		assertTrue(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.255")));
84  
85  		// test addresses that are just to the left of the block
86  		assertFalse(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.27.0")));
87  		assertFalse(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.27.124")));
88  		assertFalse(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.27.255")));
89  
90  		// test addresses that are just to the right of the block
91  		assertFalse(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.29.0")));
92  		assertFalse(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.29.124")));
93  		assertFalse(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.29.255")));
94  
95  		// test some addresses that are obviously way outside of the block
96  		assertFalse(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("10.0.0.0")));
97  		assertFalse(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("192.168.0.1")));
98  		assertFalse(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("127.0.0.1")));
99  		assertFalse(middle.isAddressMemberOfRoutingBlock(InetAddress.getByName("207.71.25.66")));
100 
101 		// for the unusualPrefixLength range, only addresses with odd value in
102 		// final octet are members
103 		assertFalse(unusualPrefixLength.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.0")));
104 		assertFalse(unusualPrefixLength.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.1")));
105 		assertFalse(unusualPrefixLength.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.2")));
106 		assertTrue(unusualPrefixLength.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.129")));
107 		assertTrue(unusualPrefixLength.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.254")));
108 		assertTrue(unusualPrefixLength.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.255")));
109 
110 		// only one address is a valid member of the "right edge" block
111 		assertFalse(rightEdge.isAddressMemberOfRoutingBlock(InetAddress.getByName("10.0.0.0")));
112 		assertFalse(rightEdge.isAddressMemberOfRoutingBlock(InetAddress.getByName("192.168.0.1")));
113 		assertFalse(rightEdge.isAddressMemberOfRoutingBlock(InetAddress.getByName("208.130.28.0")));
114 		assertFalse(rightEdge.isAddressMemberOfRoutingBlock(InetAddress.getByName("127.0.0.1")));
115 		assertTrue(rightEdge.isAddressMemberOfRoutingBlock(InetAddress.getByName("207.71.25.66")));

117 	}

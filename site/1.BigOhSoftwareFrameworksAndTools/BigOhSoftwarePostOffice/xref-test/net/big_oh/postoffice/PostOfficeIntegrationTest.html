42  /**
43   * <p>
44   * An integration test used to confirm whether a default instance of the
45   * {@link IPostOfficeService} can send email via an external SMTP server.
46   * </p>
47   * 
48   * <p>
49   * In general, details of the SMTP server used by this test are define in the postoffice.properties file.  The only exception
50   * is that the test expects to find the related password defined in a system property.  If you're executing this test from within an IDE,
51   * you'll likely need to add a JVM argument to your run configuration:
52   * <pre>-Dpostoffice.smtpPassword=super_secret_password</pre>
53   * Alternatively, you might be executing this test as part of the Maven build lifcycle, in which case you'll
54   * want to define the needed password in your settings.xml file:
55   * <pre>
56   * &lt;profile&gt;
57   *	 &lt;id&gt;BigOhSoftware&lt;/id&gt;
58   * 	 &lt;activation&gt;
59   * 		&lt;activeByDefault&gt;true&lt;/activeByDefault&gt;
60   *	 &lt;/activation&gt;
61   *	 &lt;properties&gt;
62   * 		&lt;postoffice.smtpPassword&gt;super_secret_password&lt;/postoffice.smtpPassword&gt;
63   *	 &lt;/properties&gt;
64   * &lt;/profile&gt;
65   * </pre>
66   * </p>
67   *

69   * @version Jun 2, 2010

91

94  
95  		postOfficeProperties.setProperty(PostOfficeServiceFactory.PROPERTY_KEY_SMTP_PASSWORD, smtpPassword);

97  		postOfficeService = PostOfficeServiceFactory.getInstance(postOfficeProperties);
98  
99  	}
